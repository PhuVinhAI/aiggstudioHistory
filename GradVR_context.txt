import { useState, useEffect } from 'react';
import { motion } from 'framer-motion';
import { 
  CaptureButton, VideoButton, ResetButton, SettingsButton, 
  SettingsBottomSheet, CaptureFlash, MediaPreview, TopStatusBar, ScanningGuide
} from './ui';
import { theme } from '../theme';
import { useARSettings } from '../hooks/useARSettings';
import { useARCaptureMedia } from '../hooks/useARCaptureMedia';

interface ARControlsProps {
  targetFound: boolean;
  isLoading?: boolean;
  onCapture?: (imageBlob: Blob) => void;
  onRecordStop?: (videoBlob: Blob) => void;
}

export const ARControls: React.FC<ARControlsProps> = ({
  targetFound,
  isLoading = false,
  onCapture,
  onRecordStop,
}) => {
  const [showSettings, setShowSettings] = useState(false);
  const [hasDetectedOnce, setHasDetectedOnce] = useState(false);

  useEffect(() => {
    if (targetFound && !hasDetectedOnce) {
      setHasDetectedOnce(true);
    }
  }, [targetFound, hasDetectedOnce]);
  
  const {
    enableAudio, setEnableAudio,
    enableARSound, setEnableARSound,
    enableSlamPersistence, setEnableSlamPersistence,
    useAbsoluteScale, setUseAbsoluteScale
  } = useARSettings();

  const {
    mode, isRecording, isProcessingVideo, isConverting, showFlash, preview,
    handleCapture, handleVideoClick, handleSavePreview, handleDiscardPreview
  } = useARCaptureMedia(enableAudio, onCapture, onRecordStop);

  if (isLoading) return null;

  const handleReset = () => {
    window.dispatchEvent(new CustomEvent('resetSlam'));
  };

  return (
    <motion.div
      initial={{ opacity: 0 }}
      animate={{ opacity: 1 }}
      transition={{ duration: 0.5 }}
    >
      <CaptureFlash isVisible={showFlash} />

      <TopStatusBar 
        isProcessingVideo={isProcessingVideo}
        isRecording={isRecording}
        isConverting={isConverting}
      />

      <ScanningGuide isVisible={!hasDetectedOnce && !targetFound} />

      <MediaPreview 
        preview={preview}
        onSave={handleSavePreview}
        onDiscard={handleDiscardPreview}
      />

      <div
        style={{
          position: 'fixed',
          bottom: theme.spacing.xxl,
          left: '50%',
          transform: 'translateX(-50%)',
          display: 'flex',
          alignItems: 'center',
          gap: theme.spacing.md,
          zIndex: theme.zIndex.controls,
        }}
      >
        <ResetButton onClick={handleReset} />

        <motion.div
           animate={{ opacity: (isProcessingVideo || isConverting) ? 0.3 : 1 }}
           style={{
             pointerEvents: (isProcessingVideo || isConverting || preview) ? 'none' : 'auto',
             background: theme.colors.surface.glass,
             backdropFilter: theme.backdropFilter.blurStrong,
             borderRadius: theme.borderRadius.pill,
             padding: theme.spacing.xs,
             display: 'flex',
             alignItems: 'center',
             position: 'relative',
             boxShadow: theme.shadows.md,
             border: `1px solid ${theme.colors.border.medium}`,
           }}
        >
          <motion.div
             animate={{ x: mode === 'photo' ? 0 : 68 }}
             transition={{ type: "spring", stiffness: 400, damping: 30 }}
             style={{
               position: 'absolute',
               top: theme.spacing.xs, left: theme.spacing.xs,
               width: '64px', height: '52px',
               borderRadius: '26px',
               background: theme.colors.primary,
               zIndex: 0
             }}
          />

             <CaptureButton

import { useEffect, useRef, useState } from "react";
import * as THREE from "three";
import { createXR8Scene, cleanupXR8Scene, resetXR8Scene } from "../utils/xr8Setup";
import { ARControls } from "./ARControls";
import { useXREvents } from "../hooks/useXREvents";

if (typeof window !== 'undefined') {
  (window as any).THREE = THREE;
}

interface ARSceneProps {
  targetSrc: string;
}

export default function ARScene({ targetSrc }: ARSceneProps) {
  const [targetFound, setTargetFound] = useState(false);
  const sceneSetupRef = useRef<Awaited<ReturnType<typeof createXR8Scene>> | null>(null);
  
  // Custom hook manages XRExtras events like AlmostThere and Loading screens
  const { isLoading } = useXREvents();

  const [enableSlamPersistence, setEnableSlamPersistence] = useState(() => {
    const saved = localStorage.getItem('enableSlamPersistence');
    return saved !== null ? saved === 'true' : true;
  });

  const [enableARSound, setEnableARSound] = useState(() => {
    const saved = localStorage.getItem('enableARSound');
    return saved !== null ? saved === 'true' : true;
  });

  useEffect(() => {
    const handleSettingsChange = (e: Event) => {
      const customEvent = e as CustomEvent;
      const { setting, value } = customEvent.detail;
      
      if (setting === 'enableSlamPersistence') {
        setEnableSlamPersistence(value);
      } else if (setting === 'enableARSound') {
        setEnableARSound(value);
      }
    };

    window.addEventListener('settingsChanged', handleSettingsChange);
    return () => window.removeEventListener('settingsChanged', handleSettingsChange);
  }, []);

  useEffect(() => {
    const canvas = document.createElement('canvas');
    canvas.id = 'camerafeed';
    document.body.appendChild(canvas);
    
    const handleResize = () => {
      if (sceneSetupRef.current?.renderer) {
        sceneSetupRef.current.renderer.setSize(window.innerWidth, window.innerHeight);
      }
    };
    
    window.addEventListener('resize', handleResize);
    
    const handleResetSlam = () => {
      if (sceneSetupRef.current) {
        resetXR8Scene(sceneSetupRef.current);
        setTargetFound(false);
      }
    };
    
    window.addEventListener('resetSlam', handleResetSlam);
    
    const initScene = async () => {
      try {
        await Promise.all([
          new Promise<void>((resolve) => {
            if (window.XR8) resolve();
            else window.addEventListener('xrloaded', () => resolve(), { once: true });
          }),
          new Promise<void>((resolve) => {
            if (window.XRExtras) resolve();
            else window.addEventListener('xrextrasloaded', () => resolve(), { once: true });
          }),
        ]);

        const savedSlamPersistence = localStorage.getItem('enableSlamPersistence');
        const initialSlamPersistence = savedSlamPersistence !== null ? savedSlamPersistence === 'true' : true;
        
        const savedAbsoluteScale = localStorage.getItem('useAbsoluteScale');
        const initialAbsoluteScale = savedAbsoluteScale !== null ? savedAbsoluteScale === 'true' : true;

        sceneSetupRef.current = await createXR8Scene(
          canvas,
          targetSrc,
          () => setTargetFound(true),
          () => setTargetFound(false),
          initialSlamPersistence,
          initialAbsoluteScale
        );
        
        handleResize();
      } catch (error) {
        console.error("Error initializing XR8 scene:", error);
      }
    };

    initScene();

    return () => {
      window.removeEventListener('resize', handleResize);
      window.removeEventListener('resetSlam', handleResetSlam);
      
      if (sceneSetupRef.current) cleanupXR8Scene(sceneSetupRef.current);
      if (canvas.parentNode) canvas.parentNode.removeChild(canvas);
    };
  }, [targetSrc]);

  useEffect(() => {
    if (sceneSetupRef.current?.audio) {
      if (enableARSound && targetFound) {
        sceneSetupRef.current.audio.play().catch(() => {});
      } else {
        sceneSetupRef.current.audio.pause();
      }
    }
  }, [targetFound, enableARSound]);

  useEffect(() => {
    if (sceneSetupRef.current?.slamPersistence) {
      sceneSetupRef.current.slamPersistence.setEnabled(enableSlamPersistence);
      if (!enableSlamPersistence && !targetFound && sceneSetupRef.current.anchorGroup) {
        sceneSetupRef.current.anchorGroup.visible = false;
      }
    }
  }, [enableSlamPersistence, targetFound]);

  const handleRecordStop = (blob: Blob) => {
    console.log('Video recorded:', blob.size, 'bytes');
  };

  const handleCapture = (blob: Blob) => {
    console.log('Photo captured:', blob.size, 'bytes');
  };

  return (
    <ARControls
      targetFound={targetFound}
      isLoading={isLoading}
      onCapture={handleCapture}
      onRecordStop={handleRecordStop}
    />
  );
}

import * as THREE from "three";
import { FBXLoader } from "three/addons/loaders/FBXLoader.js";

export class ChibiCharacter {
  private group: THREE.Group;
  private character: THREE.Object3D | null = null;
  private mixer: THREE.AnimationMixer | null = null;
  private mainAction: THREE.AnimationAction | null = null;
  private characterHeight = 0;

  constructor() {
    this.group = new THREE.Group();
  }

  getGroup(): THREE.Group {
    return this.group;
  }

  getHeadPosition(): THREE.Vector3 | null {
    if (!this.character || this.characterHeight === 0) return null;

    // Vị trí trên đầu nhân vật
    const pos = new THREE.Vector3(0, 0, 0);
    pos.z = this.characterHeight + 1.0;
    return pos;
  }

  load(position: THREE.Vector3): Promise<void> {
    return new Promise((resolve, reject) => {
      const loader = new FBXLoader();
      loader.load(
        `${import.meta.env.BASE_URL}models/dancing_character.fbx`,
        (fbx) => {
          this.character = fbx;
          
          // FBX từ Mixamo thường rất to, scale nhỏ lại
          // Với absolute scale: 0.005 = ~0.3-0.5m (chibi size)
          // Nếu muốn to hơn: tăng lên 0.01 = ~0.6-1m
          this.character.scale.setScalar(0.005);

          // Xoay để nhân vật đứng đúng hướng trên mặt phẳng ảnh
          this.character.rotation.x = -Math.PI / 2;
          this.character.rotation.y = Math.PI;
          this.character.rotation.z = Math.PI;

          this.character.position.copy(position);
          this.character.position.z = 0.01; // 1cm phía trên bề mặt

          // Tính chiều cao nhân vật
          const box = new THREE.Box3().setFromObject(this.character);
          this.characterHeight = box.max.z - box.min.z;

          this.group.add(this.character);

          // Setup animation từ FBX
          if (fbx.animations && fbx.animations.length > 0) {
            this.mixer = new THREE.AnimationMixer(this.character);
            const action = this.mixer.clipAction(fbx.animations[0]);
            action.setLoop(THREE.LoopRepeat, Infinity);
            action.play();
            this.mainAction = action;
          }

          resolve();
        },
        undefined,
        (error) => {
          console.error("Error loading dancing character:", error);
          reject(error);
        }
      );
    });
  }

  resetAnimation() {
    if (this.mainAction) {
      this.mainAction.reset();
      this.mainAction.play();
    }
  }

  update(deltaTime: number) {
    if (this.mixer) {
      this.mixer.update(deltaTime);
    }
  }

  dispose() {
    if (this.character) {
      this.group.remove(this.character);
      this.character.traverse((child) => {
        if (child instanceof THREE.Mesh) {
          child.geometry.dispose();
          if (Array.isArray(child.material)) {
            child.material.forEach((m) => m.dispose());
          } else {
            child.material.dispose();
          }
        }
      });
    }
    if (this.mixer) {
      this.mixer.stopAllAction();
    }
  }
}

    this.bannerMesh = new THREE.Mesh(geometry, material);

    // Xoay banner để nghiêng về phía camera nhiều hơn
    this.bannerMesh.rotation.x = Math.PI / 3; // Nghiêng khoảng 60 độ

    this.bannerMesh.position.copy(this.basePosition);
    this.group.add(this.bannerMesh);
  }

  resetAnimation() {
    this.animationTime = 0;
  }

  update(deltaTime: number) {
    this.animationTime += deltaTime;

    if (this.bannerMesh) {
      // Lơ lửng nhẹ trên đầu nhân vật (theo trục Z)
      this.bannerMesh.position.z =
        this.basePosition.z + Math.sin(this.animationTime * 1.5) * 0.02;
    }
  }

  dispose() {
    if (this.bannerMesh) {
      this.group.remove(this.bannerMesh);
      this.bannerMesh.geometry.dispose();
      (this.bannerMesh.material as THREE.MeshBasicMaterial).dispose();
    }
  }
}

import * as THREE from "three";

interface Particle {
  sprite: THREE.Sprite;
  vx: number;
  vy: number;
  vz: number;
  life: number;
}

export class Fireworks {
  private group: THREE.Group;
  private particles: Particle[] = [];
  private texture: THREE.CanvasTexture;

  constructor() {
    this.group = new THREE.Group();
    this.texture = this.createTexture();
  }

  getGroup(): THREE.Group {
    return this.group;
  }

  // Texture nhỏ, sáng, mờ dần ra ngoài
  private createTexture(): THREE.CanvasTexture {
    const canvas = document.createElement("canvas");
    canvas.width = 16;
    canvas.height = 16;
    const ctx = canvas.getContext("2d")!;

    const gradient = ctx.createRadialGradient(8, 8, 0, 8, 8, 8);
    gradient.addColorStop(0, "rgba(255,255,255,1)");
    gradient.addColorStop(0.1, "rgba(255,255,255,0.8)");
    gradient.addColorStop(0.5, "rgba(255,255,255,0.3)");
    gradient.addColorStop(1, "rgba(255,255,255,0)");

    ctx.fillStyle = gradient;
    ctx.fillRect(0, 0, 16, 16);

    return new THREE.CanvasTexture(canvas);
  }

  private createMaterial(color: number): THREE.SpriteMaterial {
    return new THREE.SpriteMaterial({
      map: this.texture,
      color: color,
      transparent: true,
      blending: THREE.AdditiveBlending,
      depthWrite: false,
    });
  }

  launch(startPosition: THREE.Vector3, color: number) {
    const targetZ = startPosition.z + 0.45 + Math.random() * 0.15;

    // Rocket nhỏ bay lên (theo trục Z = vuông góc với ảnh)
    const rocket = new THREE.Sprite(this.createMaterial(0xffffaa));
    rocket.scale.setScalar(0.025);
    rocket.position.copy(startPosition);
    this.group.add(rocket);

    let currentZ = startPosition.z;
    const flyUp = () => {
      currentZ += 0.015;
      rocket.position.z = currentZ;

      // Trail nhỏ
      const trail = new THREE.Sprite(this.createMaterial(0xffaa44));
      trail.scale.setScalar(0.015);
      trail.position.set(
        rocket.position.x + (Math.random() - 0.5) * 0.005,
        rocket.position.y,
        rocket.position.z - 0.01
      );
      this.group.add(trail);
      this.particles.push({
        sprite: trail,
        vx: 0,
        vy: 0,
        vz: -0.001,
        life: 0.3,
      });

      if (currentZ >= targetZ) {
        this.group.remove(rocket);
        rocket.material.dispose();
        this.explode(
          new THREE.Vector3(startPosition.x, startPosition.y, targetZ),
          color
        );
      } else {
        requestAnimationFrame(flyUp);
      }
    };
    requestAnimationFrame(flyUp);
  }

  private explode(position: THREE.Vector3, color: number) {
    // Rất nhiều hạt nhỏ li ti
    const numParticles = 150;

    for (let i = 0; i < numParticles; i++) {
      // Phân bố đều trên hình cầu
      const theta = Math.random() * Math.PI * 2;
      const phi = Math.acos(2 * Math.random() - 1);

      // Tốc độ có chút random để tự nhiên hơn
      const baseSpeed = 0.015;
      const speed = baseSpeed + Math.random() * 0.008;

      const sprite = new THREE.Sprite(this.createMaterial(color));
      // Hạt vừa phải
      sprite.scale.setScalar(0.012 + Math.random() * 0.008);
      sprite.position.copy(position);
      this.group.add(sprite);

      this.particles.push({
        sprite,
        vx: Math.sin(phi) * Math.cos(theta) * speed,
        vy: Math.sin(phi) * Math.sin(theta) * speed,
        vz: Math.cos(phi) * speed * 0.5,
        life: 0.8 + Math.random() * 0.6,
      });
    }

    // Thêm vài hạt sáng hơn ở giữa
    for (let i = 0; i < 20; i++) {
      const theta = Math.random() * Math.PI * 2;
      const phi = Math.acos(2 * Math.random() - 1);
      const speed = 0.008 + Math.random() * 0.005;

      const sprite = new THREE.Sprite(this.createMaterial(0xffffff));
      sprite.scale.setScalar(0.018);
      sprite.position.copy(position);
      this.group.add(sprite);

      this.particles.push({
        sprite,
        vx: Math.sin(phi) * Math.cos(theta) * speed,
        vy: Math.sin(phi) * Math.sin(theta) * speed,
        vz: Math.cos(phi) * speed * 0.5,
        life: 0.5 + Math.random() * 0.3,
      });
    }
  }

  launchSequence(centerPosition: THREE.Vector3) {
    const colors = [0xff3333, 0xffdd33, 0x33ff66, 0x33ddff, 0xff33ff, 0xff8833];

    // Bắn từ dưới ảnh (y âm = phía dưới)
    const launches = [
      { x: 0, y: -0.4, delay: 0 },
      { x: -0.18, y: -0.45, delay: 400 },
      { x: 0.18, y: -0.45, delay: 800 },
      { x: -0.08, y: -0.42, delay: 1200 },
      { x: 0.08, y: -0.42, delay: 1600 },
      { x: 0, y: -0.4, delay: 2000 },
    ];

    launches.forEach(({ x, y, delay }, i) => {
      setTimeout(() => {
        this.launch(
          new THREE.Vector3(
            centerPosition.x + x,
            centerPosition.y + y,
            0.01 // Bắt đầu từ mặt ảnh
          ),
          colors[i % colors.length]
        );
      }, delay);
    });
  }

  update() {
    const gravity = 0.0005;

    for (let i = this.particles.length - 1; i >= 0; i--) {
      const p = this.particles[i];

      p.sprite.position.x += p.vx;
      p.sprite.position.y += p.vy;
      p.sprite.position.z += p.vz;

      // Gravity kéo xuống (theo trục Z)
      p.vz -= gravity;

      // Drag - chậm dần
      p.vx *= 0.98;
      p.vy *= 0.98;
      p.vz *= 0.98;

      p.life -= 0.018;

      // Fade out
      const alpha = Math.max(0, p.life);
      (p.sprite.material as THREE.SpriteMaterial).opacity = alpha;

      // Thu nhỏ dần
      const currentScale = p.sprite.scale.x;
      p.sprite.scale.setScalar(currentScale * 0.995);

      if (p.life <= 0) {
        this.group.remove(p.sprite);
        p.sprite.material.dispose();
        this.particles.splice(i, 1);
      }
    }
  }

  dispose() {
    this.particles.forEach((p) => {
      this.group.remove(p.sprite);
      p.sprite.material.dispose();
    });
    this.particles = [];
    this.texture.dispose();
  }
}

import { useState } from 'react';
import { getMediaRecorder } from '../utils/customMediaRecorder';
import { convertToMP4, needsConversion } from '../utils/videoConverter';

export type CaptureMode = 'photo' | 'video';

export const useARCaptureMedia = (
  enableAudio: boolean,
  onCapture?: (blob: Blob) => void,
  onRecordStop?: (blob: Blob) => void
) => {
  const [mode, setMode] = useState<CaptureMode>('photo');
  const [isRecording, setIsRecording] = useState(false);
  const [isProcessingVideo, setIsProcessingVideo] = useState(false);
  const [showFlash, setShowFlash] = useState(false);
  const [preview, setPreview] = useState<{ type: 'photo' | 'video', url: string, blob: Blob } | null>(null);
  const [isConverting, setIsConverting] = useState(false);

  const mediaRecorder = getMediaRecorder();

  const handleCapture = () => {
    if (mode !== 'photo') {
      setMode('photo');
      return;
    }
    if (preview) return;

    const canvas = document.getElementById('camerafeed') as HTMLCanvasElement;
    if (!canvas) return;

    setShowFlash(true);
    const dataUrl = canvas.toDataURL('image/png');
    
    setPreview({ type: 'photo', url: dataUrl, blob: new Blob() });
    setShowFlash(false);

    fetch(dataUrl)
      .then(res => res.blob())
      .then(blob => setPreview(prev => prev ? { ...prev, blob } : null))
      .catch(error => console.error('[Photo] Error creating blob:', error));
  };

  const handleVideoClick = async () => {
    if (preview) return;

    if (isRecording) {
      mediaRecorder.stopRecording();
      setIsRecording(false);
      setIsProcessingVideo(true);
      return;
    }
    
    if (mode !== 'video') {
      setMode('video');
      return;
    }

    const canvas = document.getElementById('camerafeed') as HTMLCanvasElement;
    if (!canvas) return;

    try {
      await mediaRecorder.startRecording({
        canvas,
        fps: 30,
        videoBitsPerSecond: 8000000,
        includeAudio: enableAudio,
        onStart: () => setIsRecording(true),
        onStop: () => setIsProcessingVideo(false),
        onDataAvailable: async (blob: Blob) => {
          const url = URL.createObjectURL(blob);
          setPreview({ type: 'video', url, blob });
          
          // Auto-convert to MP4 in background nếu cần
          if (needsConversion(blob)) {
            console.log('[Video] Auto-converting to MP4...');
            setIsConverting(true);
            try {
              const mp4Blob = await convertToMP4(blob);
              const mp4Url = URL.createObjectURL(mp4Blob);
              
              // Update preview với MP4 blob
              setPreview({ type: 'video', url: mp4Url, blob: mp4Blob });
              
              // Cleanup old URL
              URL.revokeObjectURL(url);
              
              console.log('[Video] Conversion completed');
            } catch (error) {
              console.error('[Video] Conversion failed:', error);
            } finally {
              setIsConverting(false);
            }
          }
        },
        onError: (error: Error) => {
          console.error('[Video] Recording error:', error);
          setIsRecording(false);
          setIsProcessingVideo(false);
        },
      });
    } catch (error) {
      console.error('[Video] Failed to start recording:', error);
      setIsRecording(false);
      setIsProcessingVideo(false);
    }
  };

  const handleSavePreview = () => {
    if (!preview) return;
    
    if (preview.type === 'photo') {
      onCapture?.(preview.blob);
    } else {
      onRecordStop?.(preview.blob);
    }

    const a = document.createElement('a');
    if (preview.blob.size > 0) {
      const blobUrl = URL.createObjectURL(preview.blob);
      a.href = blobUrl;
      
      // LUÔN dùng PNG cho ảnh và MP4 cho video
      const extension = preview.type === 'photo' ? 'png' : 'mp4';
      a.download = `grad_ar_${Date.now()}.${extension}`;
      
      console.log('[Save] Downloading:', { 
        type: preview.type, 
        blobType: preview.blob.type, 
        extension, 
        size: preview.blob.size 
      });
      
      document.body.appendChild(a);
      a.click();
      document.body.removeChild(a);
      setTimeout(() => URL.revokeObjectURL(blobUrl), 100);
    } else {
      a.href = preview.url;
      const extension = preview.type === 'photo' ? 'png' : 'mp4';
      a.download = `grad_ar_${Date.now()}.${extension}`;
      document.body.appendChild(a);
      a.click();
      document.body.removeChild(a);
    }

    setPreview(null);
  };

  const handleDiscardPreview = () => {
    setPreview(null);
  };

  return {
    mode, setMode,
    isRecording, isProcessingVideo, isConverting,
    showFlash, preview,
    handleCapture, handleVideoClick,
    handleSavePreview, handleDiscardPreview
  };
};

import { useState, useEffect } from 'react';

export const useARSettings = () => {
  const [enableAudio, setEnableAudio] = useState(false);
  
  const [enableARSound, setEnableARSound] = useState(() => {
    const saved = localStorage.getItem('enableARSound');
    return saved !== null ? saved === 'true' : true;

  });

import { useState, useEffect } from 'react';

/**
 * Custom hook to manage recording timer state.
 * @param isRecording Boolean indicating if a recording is currently active.
 * @returns The current recording duration in seconds.
 */
export const useRecordingTimer = (isRecording: boolean) => {
  const [recordingSeconds, setRecordingSeconds] = useState(0);

  useEffect(() => {
    if (!isRecording) {
      return;
    }

    setRecordingSeconds(0);

    const interval = setInterval(() => {
      setRecordingSeconds(s => s + 1);
    }, 1000);

    return () => clearInterval(interval);
  }, [isRecording]);

  return recordingSeconds;
};

import { useEffect, useState } from 'react';
import { addXRExtrasEventListener, XRExtrasEvents } from '../utils/xrExtrasEvents';

export const useXREvents = () => {
  const [isLoading, setIsLoading] = useState(true);

  useEffect(() => {
    const checkVisibility = () => {
      setTimeout(() => {
        const almostThereContainer = document.getElementById('almostthereContainer');
        const loadingContainer = document.getElementById('loadingContainer');
        
        const isAlmostThereHidden = !almostThereContainer || almostThereContainer.classList.contains('hidden');
        const isLoadingHidden = !loadingContainer || loadingContainer.classList.contains('hidden');
        
        if (isAlmostThereHidden && isLoadingHidden) {
          setIsLoading(false);
        } else {
          setIsLoading(true);
        }
      }, 100);
    };

    const removeLoadingShown = addXRExtrasEventListener(XRExtrasEvents.LOADING_SHOWN, () => setIsLoading(true));
    const removeLoadingHidden = addXRExtrasEventListener(XRExtrasEvents.LOADING_HIDDEN, checkVisibility);
    
    const removeAlmostThereShown = addXRExtrasEventListener(XRExtrasEvents.ALMOST_THERE_SHOWN, () => setIsLoading(true));
    const removeAlmostThereHidden = addXRExtrasEventListener(XRExtrasEvents.ALMOST_THERE_HIDDEN, checkVisibility);

    return () => {
      removeLoadingShown?.();
      removeLoadingHidden?.();
      removeAlmostThereShown?.();
      removeAlmostThereHidden?.();
    };
  }, []);

  return { isLoading, setIsLoading };
};

# Theme System - GradAR

File theme chung cho toàn bộ dự án AR, giúp duy trì tính nhất quán về màu sắc, khoảng cách, và style.

## Cách sử dụng

```typescript
import { theme } from '../theme';

// Sử dụng trong style
style={{
  background: theme.colors.primary,
  padding: theme.spacing.md,
  borderRadius: theme.borderRadius.lg,
}}
```

## Cấu trúc Theme

### Colors (Màu sắc)
- `primary`: Màu chính (#ef5350 - đỏ)
- `secondary`: Màu phụ (#ff00cc - hồng)
- `background`: Màu nền (dark, overlay, overlayDark, overlayLight)
- `surface`: Màu bề mặt (white, light, glass, glassDark, glassLight)
- `text`: Màu chữ (primary, secondary, tertiary, disabled, white)
- `border`: Màu viền (light, medium)

- `xs`: 4px

/**
 * Custom MediaRecorder implementation
 * Thay thế XR8.MediaRecorder để có control tốt hơn về FPS và quality
 */

interface RecordingOptions {
  canvas: HTMLCanvasElement;
  fps?: number;
  videoBitsPerSecond?: number;
  audioBitsPerSecond?: number;
  includeAudio?: boolean;
  onStart?: () => void;
  onStop?: () => void;
  onDataAvailable?: (blob: Blob) => void;
  onError?: (error: Error) => void;
}

export class CustomMediaRecorder {
  private mediaRecorder: MediaRecorder | null = null;
  private recordedChunks: Blob[] = [];
  private stream: MediaStream | null = null;
  private audioStream: MediaStream | null = null;

  async startRecording(options: RecordingOptions): Promise<void> {
    const {
      canvas,
      fps = 30,
      videoBitsPerSecond = 8000000, // 8 Mbps
      audioBitsPerSecond = 128000, // 128 kbps
      includeAudio = false,
      onStart,
      onStop,
      onDataAvailable,
      onError,
    } = options;

    try {
      // Capture canvas stream với FPS cố định
      this.stream = canvas.captureStream(fps);
      console.log('[CustomMediaRecorder] Canvas stream captured at', fps, 'FPS');

      // Thêm audio track nếu cần
      if (includeAudio) {
        try {
          this.audioStream = await navigator.mediaDevices.getUserMedia({ 
            audio: {
              echoCancellation: true,
              noiseSuppression: true,
              autoGainControl: true,
            } 
          });
          
          // Add audio tracks vào video stream
          this.audioStream.getAudioTracks().forEach(track => {
            this.stream!.addTrack(track);
          });
          
          console.log('[CustomMediaRecorder] Audio track added');
        } catch (audioError) {
          console.warn('[CustomMediaRecorder] Failed to get audio:', audioError);
          // Continue without audio
        }
      }

      // Chọn mimeType tốt nhất
      const mimeType = this.getBestMimeType();
      console.log('[CustomMediaRecorder] Using mimeType:', mimeType);

      // Tạo MediaRecorder
      this.mediaRecorder = new MediaRecorder(this.stream, {
        mimeType,
        videoBitsPerSecond,
        audioBitsPerSecond,
      });

      this.recordedChunks = [];

      // Event handlers
      this.mediaRecorder.ondataavailable = (event) => {
        if (event.data && event.data.size > 0) {
          this.recordedChunks.push(event.data);
        }
      };

      this.mediaRecorder.onstart = () => {
        console.log('[CustomMediaRecorder] Recording started');
        onStart?.();
      };

      this.mediaRecorder.onstop = () => {
        console.log('[CustomMediaRecorder] Recording stopped');
        
        // Tạo blob từ chunks với MIME type chính xác
        const blob = new Blob(this.recordedChunks, { type: mimeType });
        console.log('[CustomMediaRecorder] Video blob created:', {
          size: blob.size,
          type: blob.type,
          mimeType: mimeType
        });
        
        onDataAvailable?.(blob);
        onStop?.();
        
        // Cleanup
        this.cleanup();
      };

      this.mediaRecorder.onerror = (event: Event) => {
        const error = new Error('MediaRecorder error: ' + (event as any).error);
        console.error('[CustomMediaRecorder]', error);
        onError?.(error);
        this.cleanup();
      };

      // Start recording
      this.mediaRecorder.start(100); // Collect data every 100ms
      
    } catch (error) {
      console.error('[CustomMediaRecorder] Failed to start recording:', error);
      onError?.(error as Error);
      this.cleanup();
      throw error;
    }
  }

  stopRecording(): void {
    if (this.mediaRecorder && this.mediaRecorder.state !== 'inactive') {
      console.log('[CustomMediaRecorder] Stopping recording...');
      this.mediaRecorder.stop();
    }
  }

  isRecording(): boolean {
    return this.mediaRecorder?.state === 'recording';
  }

  private getBestMimeType(): string {
    // Ưu tiên MP4 format để có metadata đầy đủ
    const types = [
      'video/mp4;codecs=avc1,mp4a',
      'video/mp4;codecs=h264,aac',
      'video/mp4',
      'video/webm;codecs=h264,opus',
      'video/webm;codecs=vp9,opus',
      'video/webm;codecs=vp8,opus',
      'video/webm',
    ];

    for (const type of types) {
      if (MediaRecorder.isTypeSupported(type)) {
        console.log('[CustomMediaRecorder] Selected mimeType:', type);
        return type;
      }
    }

    // Fallback
    return 'video/webm';
  }

  private cleanup(): void {
    // Stop all tracks
    if (this.stream) {
      this.stream.getTracks().forEach(track => track.stop());
      this.stream = null;
    }

    if (this.audioStream) {
      this.audioStream.getTracks().forEach(track => track.stop());
      this.audioStream = null;
    }

    this.mediaRecorder = null;
  }
}

// Singleton instance
let recorderInstance: CustomMediaRecorder | null = null;

export const getMediaRecorder = (): CustomMediaRecorder => {
  if (!recorderInstance) {
    recorderInstance = new CustomMediaRecorder();
  }
  return recorderInstance;
};

/**
 * SLAM Persistence Module - Sử dụng XR8 World Tracking đúng chuẩn
 * Giữ AR objects trong không gian thực khi mất image target
 * 
 * Cách hoạt động:
 * - XR8 có SLAM (6DoF tracking) luôn chạy ngầm
 * - Khi mất image target, ta KHÔNG ẩn object
 * - Object sẽ tự động được giữ ở vị trí trong world space nhờ SLAM
 * - AR sẽ ở lại MÃI MÃI cho đến khi tìm lại target hoặc tắt app
 */

import * as THREE from 'three';

interface SlamPersistenceConfig {
  enabled: boolean;
}

export class SlamPersistence {
  private config: SlamPersistenceConfig;
  private isTrackingLost: boolean = false;

  constructor(config?: Partial<SlamPersistenceConfig>) {
    this.config = {
      enabled: true,
      ...config,
    };
  }

  /**
   * Xử lý khi mất image target
   * KHÔNG ẩn object, để SLAM tracking tự động giữ vị trí
   */
  onTargetLost(anchorGroup: THREE.Group) {
    if (!this.config.enabled) {
      // Nếu tắt persistence, ẩn ngay
      anchorGroup.visible = false;
      return;
    }

    // Giữ nguyên visible = true
    // XR8 SLAM sẽ tự động track vị trí trong world space
    anchorGroup.visible = true;
    
    this.isTrackingLost = true;
  }

  /**
   * Xử lý khi tìm lại target
   */
  onTargetFound(anchorGroup: THREE.Group) {
    anchorGroup.visible = true;
    this.isTrackingLost = false;
  }

  /**
   * Update mỗi frame - không làm gì cả vì không có fade out
   */
  update(_anchorGroup: THREE.Group, _deltaTime: number) {
    // Không cần update gì, SLAM tự động xử lý
  }

  /**
   * Bật/tắt persistence
   */
  setEnabled(enabled: boolean) {
    this.config.enabled = enabled;
  }

  /**
   * Check xem có đang tracking với SLAM không
   */
  isUsingSlamTracking(): boolean {
    return this.isTrackingLost;
  }

  /**
   * Reset state và ẩn AR objects
   */
  reset(anchorGroup?: THREE.Group) {
    this.isTrackingLost = false;
    if (anchorGroup) {
      anchorGroup.visible = false;
    }
  }
}

/**
 * Video Converter Utility
 * Convert WebM to MP4 using FFmpeg.wasm
 */

import { FFmpeg } from '@ffmpeg/ffmpeg';
import { fetchFile, toBlobURL } from '@ffmpeg/util';

let ffmpegInstance: FFmpeg | null = null;
let isFFmpegLoaded = false;

/**
 * Load FFmpeg instance (singleton)
 */
async function loadFFmpeg(): Promise<FFmpeg> {
  if (ffmpegInstance && isFFmpegLoaded) {
    return ffmpegInstance;
  }

  if (!ffmpegInstance) {
    ffmpegInstance = new FFmpeg();
  }

  if (!isFFmpegLoaded) {
    const baseURL = 'https://unpkg.com/@ffmpeg/core@0.12.6/dist/umd';
    
    await ffmpegInstance.load({
      coreURL: await toBlobURL(`${baseURL}/ffmpeg-core.js`, 'text/javascript'),
      wasmURL: await toBlobURL(`${baseURL}/ffmpeg-core.wasm`, 'application/wasm'),
    });
    
    isFFmpegLoaded = true;
    console.log('[VideoConverter] FFmpeg loaded successfully');
  }

  return ffmpegInstance;
}

/**
 * Convert WebM blob to MP4 blob
 */
export async function convertToMP4(inputBlob: Blob): Promise<Blob> {
  // Nếu đã là MP4, return luôn
  if (inputBlob.type.includes('mp4')) {
    console.log('[VideoConverter] Already MP4, skipping conversion');
    return inputBlob;
  }

  console.log('[VideoConverter] Starting conversion...', {
    inputType: inputBlob.type,
    inputSize: inputBlob.size,
  });

  try {
    const ffmpeg = await loadFFmpeg();

    // Write input file
    const inputFileName = 'input.webm';
    const outputFileName = 'output.mp4';
    
    await ffmpeg.writeFile(inputFileName, await fetchFile(inputBlob));

    // Convert với settings tối ưu cho mobile
    await ffmpeg.exec([
      '-i', inputFileName,
      '-c:v', 'libx264',        // H.264 codec
      '-preset', 'fast',         // Fast encoding
      '-crf', '23',              // Quality (18-28, lower = better)
      '-c:a', 'aac',             // AAC audio codec
      '-b:a', '128k',            // Audio bitrate
      '-movflags', '+faststart', // Enable streaming
      outputFileName
    ]);

    // Read output file
    const data = await ffmpeg.readFile(outputFileName);
    const mp4Blob = new Blob([data as BlobPart], { type: 'video/mp4' });

    // Cleanup
    await ffmpeg.deleteFile(inputFileName);
    await ffmpeg.deleteFile(outputFileName);

    console.log('[VideoConverter] Conversion completed', {
      outputSize: mp4Blob.size,
      outputType: mp4Blob.type,
    });

    return mp4Blob;
  } catch (error) {
    console.error('[VideoConverter] Conversion failed:', error);
    // Fallback: return original blob
    return inputBlob;
  }
}

/**
 * Check if blob needs conversion
 */
export function needsConversion(blob: Blob): boolean {
  return !blob.type.includes('mp4') && blob.type.includes('video');
}

import * as THREE from 'three';
import { CongratulationText, ChibiCharacter } from '../components/effects';
import { getXRExtrasPipelineModules, initXRExtras } from './xrExtrasConfig';
import { SlamPersistence } from './slamPersistence';

interface XR8SceneSetup {
  scene: THREE.Scene;
  renderer: THREE.WebGLRenderer;
  anchorGroup: THREE.Group;
  congratText: CongratulationText;
  chibiCharacter: ChibiCharacter;
  audio: HTMLAudioElement;
  clock: THREE.Clock;
  slamPersistence: SlamPersistence;
  imageTarget: any;
  useAbsoluteScale: boolean;
}

export const createXR8Scene = async (
  canvas: HTMLCanvasElement,
  targetSrc: string,
  onTargetFound: () => void,
  onTargetLost: () => void,
  enableSlamPersistence: boolean = true,
  useAbsoluteScale: boolean = true
): Promise<XR8SceneSetup> => {
  // Wait for XR8 and XRExtras to load
  await Promise.all([
    new Promise<void>((resolve) => {
      if (window.XR8) {
        resolve();
      } else {
        window.addEventListener('xrloaded', () => resolve(), { once: true });
      }
    }),
    new Promise<void>((resolve) => {
      if (window.XRExtras) {
        resolve();
      } else {
        window.addEventListener('xrextrasloaded', () => resolve(), { once: true });
      }
    }),
  ]);

  // Initialize XRExtras
  initXRExtras();

  // Load image target
  const response = await fetch(targetSrc);
  const imageTarget = await response.json();

  if (!window.XR8) {
    throw new Error('XR8 not loaded');
  }

  window.XR8.XrController.configure({
    imageTargetData: [imageTarget],
    scale: useAbsoluteScale ? 'absolute' : 'responsive', // Sử dụng setting từ user
  });

  // Setup Three.js scene - XR8 Threejs module sẽ tự tạo
  let scene: THREE.Scene;
  let renderer: THREE.WebGLRenderer;

  // Lighting - sẽ add vào scene sau khi XR8 khởi tạo
  const ambientLight = new THREE.AmbientLight(0xffffff, 1.5);
  const directionalLight = new THREE.DirectionalLight(0xffffff, 1.0);
  directionalLight.position.set(0, 1, 2);
  
  // Performance optimization: Disable shadows (không cần cho AR)
  directionalLight.castShadow = false;

  // Audio
  const audio = new Audio(`${import.meta.env.BASE_URL}audio/celebration.mp3`);
  audio.loop = true;
  audio.volume = 0.5;

  // Create anchor group - sẽ add vào scene sau
  const anchorGroup = new THREE.Group();
  anchorGroup.visible = false;

  // Effects
  const centerPosition = new THREE.Vector3(0, 0, 0);

  const congratText = new CongratulationText();
  congratText.create();
  anchorGroup.add(congratText.getGroup());

  const chibiCharacter = new ChibiCharacter();
  await chibiCharacter.load(centerPosition);
  anchorGroup.add(chibiCharacter.getGroup());

  const clock = new THREE.Clock();

  // SLAM Persistence - Giữ AR objects khi mất target
  const slamPersistence = new SlamPersistence({
    enabled: enableSlamPersistence, // Sử dụng setting từ user
  });

  // Camera Pipeline Module
  let bannerInitialized = false;

  const cameraPipelineModule = {
    name: 'gradar-camera-pipeline',
    onStart: () => {
      // XR8 Threejs module tự động tạo scene và renderer
      const xrScene = window.XR8!.Threejs.xrScene();
      scene = xrScene.scene;
      renderer = xrScene.renderer;

      // Performance optimizations
      renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
      renderer.outputColorSpace = THREE.SRGBColorSpace;
      renderer.shadowMap.enabled = false;
      
      // Set size ngay lập tức để canvas fullscreen
      renderer.setSize(window.innerWidth, window.innerHeight);

      // Add lights và objects vào scene
      scene.add(ambientLight);
      scene.add(directionalLight);
      scene.add(anchorGroup);
    },
    onUpdate: () => {
      const deltaTime = clock.getDelta();

      if (!bannerInitialized) {
        const headPos = chibiCharacter.getHeadPosition();
        if (headPos) {
          congratText.setPosition(headPos);
          bannerInitialized = true;
        }
      }

      congratText.update(deltaTime);
      chibiCharacter.update(deltaTime);
      
      // Update SLAM persistence (xử lý fade out)
      slamPersistence.update(anchorGroup, deltaTime);
    },
    listeners: [
      {
        event: 'reality.imagefound',
        process: ({ detail }: any) => {
          if (detail.name === imageTarget.name) {
            anchorGroup.visible = true;
            // Không tự động play audio nữa, để ARScene điều khiển dựa trên enableARSound
            
            // Thông báo SLAM persistence đã tìm lại target
            slamPersistence.onTargetFound(anchorGroup);
            
            onTargetFound();
          }
        },
      },
      {
        event: 'reality.imageupdated',
        process: ({ detail }: any) => {
          if (detail.name === imageTarget.name) {
            const { position, rotation, scale } = detail;
            anchorGroup.position.copy(position);
            anchorGroup.quaternion.copy(rotation);
            anchorGroup.scale.set(scale, scale, scale);
            
            // Nếu model đang bị ẩn do resetSlam hoặc bị mất nhưng vẫn nhận được sự kiện update,
            // ta cần hiển thị lại và trigger onTargetFound ngay lập tức để người dùng không phải chờ
            if (!anchorGroup.visible) {
              anchorGroup.visible = true;
              slamPersistence.onTargetFound(anchorGroup);
              onTargetFound();
            }
          }
        },
      },
      {
        event: 'reality.imagelost',
        process: ({ detail }: any) => {
          if (detail.name === imageTarget.name) {
            // Để SLAM persistence xử lý (giữ visible, không ẩn)
            slamPersistence.onTargetLost(anchorGroup);
            
            // Tạm dừng audio - ARScene sẽ điều khiển
            
            onTargetLost();
          }
        },
      },
    ],
  };

  // Add pipeline modules with XRExtras
  const xrExtrasModules = getXRExtrasPipelineModules();
  
  window.XR8.addCameraPipelineModules([
    ...xrExtrasModules,                            // Tất cả XRExtras modules
    window.XR8.GlTextureRenderer.pipelineModule(), // Render camera feed
    window.XR8.Threejs.pipelineModule(),           // Three.js integration
    // REMOVED: XR8.MediaRecorder - dùng custom MediaRecorder thay thế
    cameraPipelineModule,                          // Custom logic - PHẢI TRƯỚC XrController
    window.XR8.XrController.pipelineModule(),      // Image tracking controller
  ]);

  // Start XR8
  await window.XR8.run({ canvas });

  // Scene và renderer đã được khởi tạo trong onStart
  return {
    scene: scene!,
    renderer: renderer!,
    anchorGroup,
    congratText,
    chibiCharacter,
    audio,
    clock,
    slamPersistence,
    imageTarget,
    useAbsoluteScale
  };
};

export const cleanupXR8Scene = (setup: XR8SceneSetup) => {
  setup.audio.pause();
  setup.audio.src = '';
  setup.congratText.dispose();
  setup.chibiCharacter.dispose();
  setup.slamPersistence.reset();

  if (window.XR8) {
    try {
      window.XR8.stop();
    } catch {
      // ignore
    }
  }
};

export const resetXR8Scene = (setup: XR8SceneSetup) => {
  // Reset SLAM persistence và ẩn AR
  setup.slamPersistence.reset(setup.anchorGroup);
  
  // Dừng audio
  setup.audio.pause();
  setup.audio.currentTime = 0;
  
  // Reset animations
  setup.chibiCharacter.resetAnimation();
  setup.congratText.resetAnimation();
  
  // Tạm thời vô hiệu hóa image tracking sau đó bật lại ngay lập tức
  // Điều này buộc XR8 phải loại bỏ trạng thái tracking cũ và dò lại hình ảnh từ đầu
  if (window.XR8 && window.XR8.XrController) {
    const xr8 = window.XR8;
    // 1. Clear targets
    xr8.XrController.configure({ imageTargetData: [] });
    
    // 2. Re-add targets after a tiny delay to ensure proper state reset
    setTimeout(() => {
      if (window.XR8) {
        window.XR8.XrController.configure({
          imageTargetData: [setup.imageTarget],
          scale: setup.useAbsoluteScale ? 'absolute' : 'responsive',
        });
        // Optionally call recenter to reset SLAM physics origin as well
        if (typeof (window.XR8.XrController as any).recenter === 'function') {
          (window.XR8.XrController as any).recenter();
        }
      }
    }, 50);
  }
};

  LOADING_SHOWN: 'xrextras-loading-shown',

} as const;

export type XRExtrasEventType = typeof XRExtrasEvents[keyof typeof XRExtrasEvents];

/**
 * Listen to XRExtras event
 * Returns cleanup function
 */
export const addXRExtrasEventListener = (
  eventType: XRExtrasEventType,
  callback: (event: CustomEvent) => void
) => {
  const handler = callback as EventListener;
  window.addEventListener(eventType, handler);
  return () => window.removeEventListener(eventType, handler);
};
